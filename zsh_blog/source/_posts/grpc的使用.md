### 下载安装
[下载](https://github.com/protocolbuffers/protobuf/releases/tag/v3.5.1)protocol buffer v3版本编译器: protoc-3.5.1-win32.zip ,解压后把protoc.exe放入 GOPATH/bin下,再放入windows/system32下

##### 下载protoc的golang插件
```bash
$ go get -u github.com/golang/protobuf/protoc-gen-go
## 不能直接访问google.golang.org网址时, 从github下载然后放到google.golang.org目录
$ mkdir -p src/google.golang.org/
$ cd src/google.golang.org
$ git clone https://github.com/google/go-genproto genprotog
```
##### 下载golang实现的grpc
```bash
## 可直接访问google.golang.org时
$ go get -u google.golang.org/grpc
 ## 同样不能访问google.golang.org是采用的方法
$ mkdir -p src/google.golang.org
$ cd src/google.golang.org
$ git clone https://github.com/grpc/grpc-go grpc
 ## 另外， grpc依赖的其他包需要一并下载
$ mkdir -p src/golang.org/x
$ cd src/golang.org/x
$ git clone https://github.com/golang/net
$ git clone https://github.com/golang/text
```

### 使用
##### 新建项目seeu_demo1
新建文件夹 pb
新建文件 Hello.proto
```bash
syntax = "proto3";

package pb;

service SayHello {
	rpc SayHello (ReqContent) returns (ResContent) {}
}

message ReqContent {
    string content = 1;
}

message ResContent {
    string content = 1;
}
```

执行命令生成对应的.go文件
```bash
$ cd /seeu_demo1/pb
$ protoc --go_out=plugins=grpc:. Hello.proto
```

##### 新建文件夹service,文件service.go
```bash
package services

import (
	"context"
	"log"
	"net"
	"zsh_demo1/proto"

	"google.golang.org/grpc"
)

const (
	port = ":50051"
)

type server struct{}

func (s *server) SayHello(ctx context.Context, in *proto.ReqContent) (*proto.ResContent, error) {
	return &proto.ResContent{Content: "Hello " + in.Content}, nil
}

func GrpcService() {
	lis, err := net.Listen("tcp", port)
	if err != nil {
		log.Fatal("failed to listen: %v", err)
	}
	s := grpc.NewServer()
	proto.RegisterSayHelloServer(s, &server{})
	s.Serve(lis)
}
```

main.go 里调用
```bash
func main() {
	services.GrpcService()
}
```

##### 新建文件夹client,文件client.go
```bash
const (
	address     = "localhost:50051"
	defaultName = "world"
)

func GrpcClient() {
	conn, err := grpc.Dial(address, grpc.WithInsecure())
	if err != nil {
		log.Fatal("did not connect: %v", err)
	}
	defer conn.Close()
	c := pb.NewSayHelloClient(conn)

	name := defaultName
	if len(os.Args) > 1 {
		name = os.Args[1]
	}
	r, err := c.SayHello(context.Background(), &pb.ReqContent{Content: name})
	if err != nil {
		log.Fatal("could not greet: %v", err)
	}
	log.Printf("Greeting: %s", r.Content)
}
```
main.go 里调用
```bash
func main() {
	client.GrpcClient()
}
```
